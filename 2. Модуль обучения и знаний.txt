class EducationModule:
    def __init__(self):
        self.learning_models = load_learning_models()

    def interact_with_user(self, user_input):
        response = self.learning_models.generate_response(user_input)
        return response

    def generate_learning_content(self, topic):
        content = self.learning_models.create_content(topic)
        return content

    def assess_user_knowledge(self, user_input):
        assessment = self.learning_models.evaluate(user_input)
        return assessment
